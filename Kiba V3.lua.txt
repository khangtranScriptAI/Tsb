local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local lp = Players.LocalPlayer
local vu5 = true

-- ==================== ANIMATIONS ====================
local Animations = {
    ["rbxassetid://98542310119798"] = 3.3,
    ["rbxassetid://95034083206292"] = 3.1,
    ["rbxassetid://12296113986"] = 1.7,
    ["rbxassetid://94638356008696"] = 3.35
}

local Character, Humanoid, RootPart
local debounce = false

-- ==================== FIND TARGET ====================
local function getNearestTarget()
    if not RootPart then
        return nil
    end

    local closest = nil
    local shortest = math.huge

    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("Model") and v ~= Character then

            local hum = v:FindFirstChildOfClass("Humanoid")
            local hrp = v:FindFirstChild("HumanoidRootPart")

            if hum and hrp and hum.Health > 0 then

                local dist =
                    (hrp.Position - RootPart.Position).Magnitude

                if dist < shortest then
                    shortest = dist
                    closest = hrp
                end
            end
        end
    end

    return closest
end

-- ==================== DELETE BODY MOVERS ====================
local function deleteBodyMovers()
    if not Character then
        return
    end

    for _, v in pairs(Character:GetDescendants()) do
        if v:IsA("BodyMover")
        or v:IsA("LinearVelocity")
        or v:IsA("AngularVelocity")
        or v:IsA("VectorForce")
        or v:IsA("AlignPosition")
        or v:IsA("AlignOrientation") then

            pcall(function()
                v:Destroy()
            end)
        end
    end
end

-- ==================== RESTORE ====================
local function restoreState(oldSpeed, oldJump)
    if Humanoid then
        Humanoid.PlatformStand = false
        Humanoid.AutoRotate = true
        Humanoid.WalkSpeed = oldSpeed
        Humanoid.JumpPower = oldJump

        Humanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
    end
end

-- ==================== DASH ====================
local function StickDash()

    if debounce then
        return
    end

    debounce = true

    if not (Character and Humanoid and RootPart) then
        debounce = false
        return
    end

    -- LOCK TARGET 1 LẦN
    local lockedTarget = getNearestTarget()

    if not lockedTarget then
        debounce = false
        return
    end

    -- SAVE
    local oldSpeed = Humanoid.WalkSpeed
    local oldJump = Humanoid.JumpPower

    -- LOCK PLAYER
    Humanoid.WalkSpeed = 0
    Humanoid.JumpPower = 0
    Humanoid.PlatformStand = true
    Humanoid.AutoRotate = false

    deleteBodyMovers()

    local start = tick()

    -- NGHIÊNG
    RootPart.CFrame =
        RootPart.CFrame *
        CFrame.Angles(math.rad(85), 0, 0)

    -- STICK TARGET
    local con
    con = RunService.Heartbeat:Connect(function()

        -- TARGET BIẾN MẤT
        if not lockedTarget
        or not lockedTarget.Parent then

            con:Disconnect()

            restoreState(oldSpeed, oldJump)
            debounce = false

            return
        end

        -- TARGET CHẾT
        local targetHum =
            lockedTarget.Parent:FindFirstChildOfClass("Humanoid")

        if not targetHum
        or targetHum.Health <= 0 then

            con:Disconnect()

            restoreState(oldSpeed, oldJump)
            debounce = false

            return
        end

        local elapsed = tick() - start

        if elapsed < 0.7 then

            local pos =
                lockedTarget.Position -
                lockedTarget.CFrame.LookVector * 0.3

            RootPart.Velocity = Vector3.zero
            RootPart.RotVelocity = Vector3.zero

            RootPart.CFrame =
                CFrame.new(pos, lockedTarget.Position) *
                CFrame.Angles(math.rad(85), 0, 0)

        else
            con:Disconnect()
        end
    end)

    -- DASH PACKET
    task.delay(0.18, function()

        local comm = Character:FindFirstChild("Communicate")

        if comm then
            pcall(function()
                comm:FireServer({
                    {
                        Dash = Enum.KeyCode.W,
                        Key = Enum.KeyCode.Q,
                        Goal = "KeyPress"
                    }
                })
            end)
        end
    end)

    -- GET UP
    task.delay(0.3, function()
        if Humanoid then
            Humanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
        end
    end)

    -- RESTORE
    task.delay(0.8, function()

        restoreState(oldSpeed, oldJump)

        debounce = false
    end)
end

-- ==================== UI ====================
local screen = Instance.new("ScreenGui")
screen.Name = "LethalDashMerged"
screen.ResetOnSpawn = false
screen.Parent = lp:WaitForChild("PlayerGui")

local button = Instance.new("TextButton")
button.Size = UDim2.new(0, 70, 0, 50)
button.Position = UDim2.new(0, 20, 0, 100)
button.BackgroundColor3 = Color3.fromRGB(0, 170, 255)
button.Text = "ON"

button.Font = Enum.Font.GothamBold
button.TextScaled = true

Instance.new("UICorner", button).CornerRadius = UDim.new(0, 8)

button.MouseButton1Click:Connect(function()

    vu5 = not vu5

    button.Text = vu5 and "ON" or "OFF"

    button.BackgroundColor3 =
        vu5 and Color3.fromRGB(0,170,255)
        or Color3.fromRGB(100,100,100)
end)

-- ==================== ANIMATION DETECT ====================
local function onAnimationPlayed(track)

    if not vu5 then return end
    if debounce then return end
    if not track.Animation then return end

    local animId = track.Animation.AnimationId
    local delayTime = Animations[animId]

    if delayTime then

        task.spawn(function()

            -- DELAY THEO ANIMATION
            task.wait(delayTime)

            if lp.Character then

                Character = lp.Character

                Humanoid =
                    Character:FindFirstChildOfClass("Humanoid")

                RootPart =
                    Character:FindFirstChild("HumanoidRootPart")

                if Humanoid and RootPart then
                    StickDash()
                end
            end
        end)
    end
end

-- ==================== CHARACTER ====================
local function setupCharacter(char)

    Character = char

    Humanoid =
        char:WaitForChild("Humanoid")

    RootPart =
        char:WaitForChild("HumanoidRootPart")

    Humanoid.AnimationPlayed:Connect(onAnimationPlayed)
end

if lp.Character then
    setupCharacter(lp.Character)
end

lp.CharacterAdded:Connect(setupCharacter)
